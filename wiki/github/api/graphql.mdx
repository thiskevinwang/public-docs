```graphql
query GetFileContents($owner: String!, $name: String!, $expression: String!) {
  rateLimit {
    used
    remaining
  }
  repository(owner: $owner, name: $name) {
    name
    object(expression: $expression) {
      __typename
      id
      ... on Blob {
        text
      }
    }
  }
}
```
